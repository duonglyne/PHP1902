LIKE (thường dùng để tìm kiếm)
        Cú pháp: SELECT column1, column2,... FROM table_name WHERE column LIKE patern;

    Ký hiệu trong MySQL :
        % đại diện cho 0,1 hoặc nhiều ký tự
        _ đại diện cho một ký tự
    ví dụ:
    WHERE CustormerName LIKE 'a%' -> tìm mọi giá trị của tên khách hàng bắt đầu bằng chữ a
    -   SELECT * FROM custormers WHERE first_name LIKE 'a%';

    WHERE CustormerName LIKE '%a' -> tìm mọi giá trị của tên khách hàng kết thúc bằng chữ a
    -   SELECT * FROM custormers WHERE first_name LIKE '%a';

    WHERE CustormerName LIKE '%or%' -> tìm mọi giá trị của tên khách hàng có chứa chữ or
    -   SELECT * FROM custormers WHERE first_name LIKE '%or%';

    WHERE CustomerName LIKE '_r%'	Tìm mọi giá trị của tên khách hàng có chữ "r" là chữ thứ 2
    -   SELECT * FROM custormers WHERE first_name LIKE '_r%';

     WHERE CustomerName LIKE 'a__%'	Tìm mọi giá trị của tên khách hàng có bắt đầu là chữ "a" và có ít nhất 3
    -   SELECT * FROM custormers WHERE first_name LIKE 'a__%';

    WHERE CustomerName LIKE 'a%o'	Tìm mọi giá trị của tên khách hàng có chữ bắt đầu là "a" là chữ
        kế thúc là "o"
    -   SELECT * FROM custormers WHERE first_name LIKE 'a%o';
